#include
using namespace std;
#include  //单端数组

void printVector(vector& v)
{
	for (vector::iterator it = v.begin(); it != v.end(); it++)
	{
		cout << *it << " ";
	}
	cout << endl;
}

//vector容器构造
void test01()
{
	vector v1;  //默认构造 无参构造

	for (int i = 0; i < 10; i++)
	{
		v1.push_back(i);
	}

	printVector(v1);

	//通过区间的方式进行构造
	vector v2(v1.begin(), v1.end());
	printVector(v2);

	//n个elem方式构造
	vector v3(10, 100);
	printVector(v3);

	//拷贝构造
	vector v4(v3);
	printVector(v4);
}

int main()
{
	test01();

	return 0;
}
