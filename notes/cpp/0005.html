05 字符串型.cpp

            
#include<iostream>
using namespace std;
#include<string>  //在用C++ 风格字符串时要包含这个头文件 

int main()
{
	//C风格字符串
	//注意事项1  char字符串名[] 
	//注意事项2  等号后面要用双引号包含起来字符串 
	char str[] = "hello world";
	cout << str << endl;
	
	//C++风格字符串
	//注意事项  包含一个头文件 #include<string>
	string str2 = "hello world";
	cout << str2 << endl;
	
	return 0;
}
