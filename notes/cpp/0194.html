//内建函数对象 逻辑仿函数
//逻辑非 logical_not

void test01()
{
	vector v;

	v.push_back(true);
	v.push_back(false);
	v.push_back(true);
	v.push_back(false);

	for (vector::iterator it = v.begin(); it != v.end(); it++)
	{
		cout << *it << " ";
	}
	cout << endl;

	//利用逻辑非 将容器v搬运到容器v2中，并执行取反操作
	vector v2;
	v2.resize(v.size());  //不开辟空间无法搬运

	transform(v.begin(), v.end(), v2.begin(), logical_not());

	for (vector::iterator it = v2.begin(); it != v2.end(); it++)
	{
		cout << *it << " ";
	}
	cout << endl;
}

int main()
{
	test01();

	return 0;
}
