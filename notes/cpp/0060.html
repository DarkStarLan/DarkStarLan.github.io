#include<iostream>
using namespace std;

int main()
{
	int arr[5] = {1, 3, 2, 5, 4};
	cout << "数组逆置前:" << endl;
	for(int i = 0;i < 5;i++)
	{
		cout << arr[i] <<endl;
	}
	
	//1.记录起始下标位置 
	//2.记录结束下标位置 
	//3.起始下标与结束下标的元素互换 
	//4.起始位置++  结束位置--
	//5.循环执行1的操作 ，直到起始位置>=结束位置 
	int start = 0;
	int end = sizeof(arr) / sizeof(arr[0]) - 1;
	
	while(start < end)
	{
		//实现元素互换 
		int temp = arr[start];
		arr[start] = arr[end];
		arr[end] = temp;
		
		//下标更新
		start++;
		end--;
	}
	
	cout << "数组元素逆置后" << endl;
	for(int i = 0;i < 5;i++)
	{
		cout << arr[i] <<endl;
	}
	
	return 0;
}
