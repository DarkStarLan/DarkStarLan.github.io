#include
using namespace std;
#include

//list容器构造函数

void printList(const list& L)
{
	for (list::const_iterator it = L.begin(); it != L.end(); it++)
	{
		cout << *it << " ";
	}
	cout << endl;
}

void test01()
{
	//创建list容器
	list L1;  //默认构造

	//添加元素
	L1.push_back(10);
	L1.push_back(20);
	L1.push_back(30);
	L1.push_back(40);

	//遍历容器
	printList(L1);

	//区间方式构造
	list L2(L1.begin(), L1.end());
	printList(L2);

	//拷贝构造
	list L3(L2);
	printList(L3);

	//n个ele
	list L4(10, 1000);
	printList(L4);
}

int main()
{
	test01();

	return 0;
}
