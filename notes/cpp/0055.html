#include<iostream>
using namespace std;
#include<stdlib.h>  //rand头文件 
#include<ctime>  //srand头文件 

int main()
{
	//添加随机数种子  利用当前系统时间 
	srand((unsigned int)time(NULL));
	
	//系统生成随机数 
	int num = rand() % 100 + 1;  //rand() % 100 生成0到99的随机数 
	
	//cout << num << endl;
	
	int val = 0;
	
	while(1)
	{
		cin >> val;
	
		if(val > num)
		{
			cout << "猜测过大" << endl;
		}
		else if(val < num)
		{
			cout << "猜测过小" << endl;
		}
		else
		{
			cout << "恭喜您猜对了" << endl;
			break;  //在循环中可以退出当前循环 
		}
	}
	
	return 0;
}
