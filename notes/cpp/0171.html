using namespace std;

//list容器赋值和交换

void printList(const list& L)
{
	for (list::const_iterator it = L.begin(); it != L.end(); it++)
	{
		cout << *it << " ";
	}
	cout << endl;
}

//赋值
void test01()
{
	list L1;

	L1.push_back(10);
	L1.push_back(20);
	L1.push_back(30);
	L1.push_back(40);

	printList(L1);

	list L2;
	L2 = L1;  //operator=赋值
	printList(L2);

	list L3;
	L3.assign(L2.begin(), L2.end());
	printList(L3);

	list L4;
	L4.assign(10, 100);
	printList(L4);

}

//交换
void test02()
{
	list L1;

	L1.push_back(10);
	L1.push_back(20);
	L1.push_back(30);
	L1.push_back(40);

	list L2;
	L2.assign(10, 100);

	cout << "交换前: " << endl;
	printList(L1);
	printList(L2);

	L1.swap(L2);
	cout << "交换后: " << endl;
	printList(L1);
	printList(L2);
}

int main()
{
	//test01();
	test02();

	return 0;
}
