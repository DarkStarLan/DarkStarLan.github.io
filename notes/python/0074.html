09 复写和使用父类成员.py

            
class Phone:
    IMEI = None  # 序列号
    producer = "ITCAST"  # 厂商

    def call_by_5g(self):
        print("使用5g网络进行通话")


# 定义子类，复写父类成员
class MyPhone(Phone):
    producer = "ITHEIMA"  # 复写父类的成员属性

    def call_by_5g(self):
        print("开启CPU单核模式...")
        # 方式1
        Phone.call_by_5g(self)  # 必须加上self
        print(f"父类的厂商是：{Phone.producer}")
        # 方式2
        super().call_by_5g()
        print(f"父类的厂商是：{super().producer}")


phone = MyPhone()
phone.call_by_5g()
print(phone.producer)
# 在子类中，调用父类成员
