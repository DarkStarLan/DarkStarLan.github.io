"""
演示文件的写入
"""

# 打开文件，不存在的文件
f = open("text.txt", "w", encoding="UTF-8")
# write写入
f.write("Hello World")

# flush刷新
f.flush()

# close关闭
f.close()  # close()内置了flush()

# 打开一个存在的文件
f = open("text.txt", "w", encoding="UTF-8")

# write写入，flush刷新
f.write("黑马程序员")

# close关闭
f.close()
