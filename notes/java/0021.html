//类和封装
//相关文件：Phone.java
package com.darkstarlan.demo02;

public class Java_04 {
    public static void main(String[] args) {
        //创建手机对象
        Phone p = new Phone();

        //给手机赋值
        p.setBrand("小米");
        p.price = 1999.98;

        //获取手机对象中的值
        System.out.println(p.getBrand());
        System.out.println(p.price);

        //调用手机中的方法
        p.call();
        p.playGame();

        Phone p2 = new Phone();
        p2.setBrand("苹果");
        p2.price = 8999;

        p2.call();
        p2.playGame();
    }
}
