//自增自减运算符
package com.darkstarlan.demo01;

public class Java_06 {
    public static void main(String[] args) {
        //== 和 --
        int a = 10;
        a++;  //表示把变量a里面的值+1
        System.out.println(a);
        ++a;  //表示把变量a里面的值+1
        System.out.println(a);
        a--;  //表示把变量a里面的值-1
        System.out.println(a);
        --a;  //表示把变量a里面的值-1
        System.out.println(a);

        System.out.println("----------");

        int x = 10;
        int y = x++;  //先用后加
        int z = ++x;  //先加后用
        System.out.println("x:" + x);
        System.out.println("y:" + y);
        System.out.println("z:" + z);
    }
}
