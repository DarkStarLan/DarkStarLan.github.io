//多态
package com.darkstarlan.demo04;

public class Java_01 {
    public static void main(String[] args) {
        //当用基类=new派生类创建对象时
        //访问的成员变量是基类的
        //调用的函数是派生类的

        //创建对象
        Student s = new Student();
        s.setName("张三");
        s.setAge(18);

        Teacher t = new Teacher();
        t.setName("老王");
        t.setAge(30);

        Administrator admin = new Administrator();
        admin.setName("管理员");
        admin.setAge(35);

        register(s);
        register(t);
        register(admin);
    }

    public static void register(Person p) {
        p.show();
    }
}
