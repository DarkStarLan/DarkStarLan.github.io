//算术运算符
package com.darkstarlan.demo01;

public class Java_02 {
    //1.整数参与计算，结果只能得到整数
    //2.小数参与计算，结果有可能不精确
    public static void main(String[] args) {
        //除法
        System.out.println(10 / 2);
        System.out.println(10 / 3);
        System.out.println(10.0 / 3);

        //取模
        System.out.println(10 % 2);
        System.out.println(10 % 3);
    }
}
