//赋值运算符
package com.darkstarlan.demo01;

public class Java_07 {
    public static void main(String[] args) {
        int a = 10;
        int b = 20;
        a += b;  //a = (int)(a + b)
        a -= b;  //a = (int)(a - b)
        System.out.println(a);
        System.out.println(b);

        //+=, -=, *=, %=底层都隐藏了一个强制类型转换
        short s = 1;
        s += 1;  //s = (short)(s + 1)
        System.out.println(s);
    }
}
