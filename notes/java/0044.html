package com.darkstarlan.demo03;

public class Son extends Father {
    public String name = "Son";

    public Son() {
        //super();  默认添加在构造函数的第一行，可不写
        System.out.println("我继承了" + super.name);
    }

    public void printFatherName() {
        System.out.println(super.name);
    }

    public void printSelfName() {
        System.out.println(this.name);
    }

    @Override  //@Override可以辅助检查重写是否正确
    public void eat() {
        System.out.println("吃毛线");
    }
}
