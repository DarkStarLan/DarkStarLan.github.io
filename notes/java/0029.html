package com.darkstarlan.demo02;

public class Phone {
    //属性
    private String brand;
    public double price;

    //行为
    public Phone()
    {
        System.out.println("Phone构造函数");
    }

    public Phone(String brand, double price)
    {
        this.brand = brand;
        this.price = price;
        System.out.println("Phone有参构造函数");
    }

    public void setBrand(String brand) {
        this.brand = brand;
    }

    public String getBrand() {
        return this.brand;
    }

    public void call()
    {
        System.out.println("手机在打电话");
    }

    public void playGame()
    {
        System.out.println("手机在玩游戏");
    }
}
