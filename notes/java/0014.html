//switch语句
package com.darkstarlan.demo01;

import java.util.Scanner;

public class Java_14 {
    public static void main(String[] args) {
        System.out.println("你吃什么");
        Scanner sc = new Scanner(System.in);
        String noodles = sc.nextLine();
        switch(noodles){
            case "兰州拉面":
                System.out.println("吃兰州拉面");
                break;
            case "武汉热干面":
                System.out.println("吃武汉热干面");
                break;
            case "北京炸酱面":
                System.out.println("吃北京炸酱面");
                break;
            case "陕西油泼面":
                System.out.println("吃陕西油泼面");
                break;
            default:
                System.out.println("吃毛线");
                break;
        }

        //另一种写法
        switch(noodles){
            case "兰州拉面" -> System.out.println("吃兰州拉面");
            case "武汉热干面" -> System.out.println("吃武汉热干面");
            case "北京炸酱面" -> System.out.println("吃北京炸酱面");
            case "陕西油泼面" -> System.out.println("吃陕西油泼面");
            default -> System.out.println("吃毛线");
        }
    }
}
