10 自增自减运算符.cs

            
﻿class Program
{
    static void Main(string[] args)
    {
        //自增自减运算符
        int number = 10;
        ++number;  //先递增再返回值
        number++;  //先返回值再递增
        --number;  //先递减再返回值
        number--;  //先返回值再递减
        Console.WriteLine(number);
        Console.ReadKey();

        number = 10;
        int result = 10 + number++;
        Console.WriteLine(number);
        Console.WriteLine(result);
        Console.ReadKey();

        int a = 5;
        int b = a++ + ++a * 2 + --a + a++;
        //       5  +  7  * 2 +  6  +  6
        Console.WriteLine(a);  //7
        Console.WriteLine(b);  //31
        Console.ReadKey();
    }
}
