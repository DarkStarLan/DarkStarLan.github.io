﻿class Program
{
    public struct Person
    {
        public string _name;  //字段(加public)
        public int _age;  //字段
        public Gender _gender;  //字段
    }

    public enum Gender
    {
        男,
        女
    }

    public struct MyColor
    {
        public int _red;
        public int _green;
        public int _blue;
    }

    static void Main(string[] args)
    {
        //XX大学管理系统
        //姓名 性别 年龄 年级
        //string zsName = "张三";
        //int zsAge = 21;
        //char zsGender = '男';
        //int zsGrade = 3;

        Person zsPerson;
        zsPerson._name = "张三";
        zsPerson._age = 21;
        zsPerson._gender = Gender.男;

        Person lsPerson;
        lsPerson._name = "李四";
        lsPerson._age = 22;
        lsPerson._gender = Gender.女;

        Console.WriteLine(zsPerson._name);
        Console.WriteLine(lsPerson._name);
        Console.ReadKey();

        MyColor mc;
        mc._red = 255;
        mc._green = 0;
        mc._blue = 0;
    }
}
