﻿class Program
{
    static void Main(string[] args)
    {
        Console.WriteLine("请输入时间");
        int mins = Convert.ToInt32(Console.ReadLine());
        if(mins > 60)
        {
            Console.WriteLine("不戳，时间够了");
        }
        else if (mins > 40)
        {
            Console.WriteLine("还可以");
        }
        else
        {
            Console.WriteLine("不行，逊");
        }
        Console.ReadKey();
    }
}
