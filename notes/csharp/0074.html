﻿public interface IFlyable
{
    //不允许有访问修饰符 默认为public
    //方法、自动属性、索引
    void Fly();
}

class Person : IFlyable
{
    public void Fly()
    {
        Console.WriteLine("人类飞飞飞");
    }
}

public interface M1
{
    void Test1();
}

public interface M2
{
    void Test2();
}

public interface M3
{
    void Test3();
}

public interface SupperInterface : M1, M2, M3
{

}

public class Car : SupperInterface
{
    public void Test1()
    {
        throw new NotImplementedException();
    }

    public void Test2()
    {
        throw new NotImplementedException();
    }

    public void Test3()
    {
        throw new NotImplementedException();
    }
}

public class Bird : IFlyable
{
    public void Fly()
    {
        Console.WriteLine("鸟类飞飞飞");
    }
}

class Program
{
    static void Main(string[] args)
    {
        //接口不能被实例化
        //接口不能有字段和构造函数
        //接口与接口之间可以多继承，并且可以多继承
        //接口只能继承接口，不能继承类
        //实现接口的派生类必须实现接口的全部成员
        //如果一个派生类类继承了基类和接口，应该把基类写在前面
        //IFlyable fly = new IFlyable();  //错误
        IFlyable fly = new Person();
        fly.Fly();  //多态
        fly = new Bird();
        fly.Fly();  //多态
        Console.ReadKey();
    }
}
