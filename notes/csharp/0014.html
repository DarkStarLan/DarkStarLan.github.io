﻿class Program
{
    static void Main(string[] args)
    {
        decimal salary = 5000m;
        bool b = true;
        Console.WriteLine("请输入年终评定");
        string level = Console.ReadLine();
        switch(level)
        {
            case "a":
                salary += 500;
                break;
            case "b":
                salary += 200;
                break;
            case "c":
                salary -= 200;
                break;
            case "d":
                salary -= 500;
                break;
            default:
                Console.WriteLine("输入有误");
                b = false;
                break;
        }
        if(b)
        {
            Console.WriteLine("明年的工资是{0}元", salary);
        }
        Console.ReadKey();
    }
}
