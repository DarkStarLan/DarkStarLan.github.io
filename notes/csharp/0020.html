﻿class Program
{
    static void Main(string[] args)
    {
        //使用Convert进行转换，失败了抛出异常
        int numberOne = Convert.ToInt32("123");

        int number = int.Parse("123");
        Console.WriteLine(number);
        Console.ReadKey();

        bool b = int.TryParse("123", out number);
        Console.WriteLine(b);
        Console.WriteLine(number);
        Console.ReadKey();
    }
}
