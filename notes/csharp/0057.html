﻿using System.Text;

class Program
{
    static void Main(string[] args)
    {
        //读取文件
        byte[] buffer = File.ReadAllBytes(@"E:\桌面\test.txt");
        string s = Encoding.Default.GetString(buffer);
        Console.WriteLine(s);
        Console.ReadKey();

        string[] contents = File.ReadAllLines(@"E:\桌面\test.txt");
        foreach (string item in contents)
        {
            Console.WriteLine(item);
        }
        Console.ReadKey();

        string str = File.ReadAllText(@"E:\桌面\test.txt");
        Console.WriteLine(str);
        Console.ReadKey();

        //写文件
        //File.WriteAllLines(@"E:\桌面\test.txt", new string[] { "123", "Hello World!" });
        //Console.ReadKey();

        //File.WriteAllText(@"E:\桌面\test.txt", "123\nHello World!");
        //Console.ReadKey();

        //File.AppendAllText(@"E:\桌面\test.txt", "!");
        //Console.ReadKey();
    }
}
