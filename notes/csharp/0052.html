17 集合的长度问题.cs

            
﻿using System.Collections;

class Program
{
    static void Main(string[] args)
    {
        ArrayList list = new ArrayList();
        //count     大小
        //capacity  容量
        Console.WriteLine("大小为: {0}", list.Count);  //0
        Console.WriteLine("容量为: {0}", list.Capacity);  //0
        Console.WriteLine("添加一个元素");
        list.Add(1);
        Console.WriteLine("大小为: {0}", list.Count);  //1
        Console.WriteLine("容量为: {0}", list.Capacity);  //4

        Console.ReadKey();
    }
}
