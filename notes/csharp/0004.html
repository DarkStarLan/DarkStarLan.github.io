04 交换变量.cs

            
﻿class Program
{
    static void Main(string[] args)
    {
        int n1 = 10;
        int n2 = 20;
        Console.WriteLine("n1 = {0}, n2 = {1}", n1, n2);

        //交换n1和n2的三种方法
        int temp = n1;
        n1 = n2;
        n2 = temp;
        Console.WriteLine("n1 = {0}, n2 = {1}", n1, n2);

        n1 = n1 - n2;
        n2 = n1 + n2;
        n1 = n2 - n1;
        Console.WriteLine("n1 = {0}, n2 = {1}", n1, n2);

        (n1, n2) = (n2, n1);
        Console.WriteLine("n1 = {0}, n2 = {1}", n1, n2);
    }
}
