﻿using System.Diagnostics;
using System.Text;

class Program
{
    static void Main(string[] args)
    {
        //字符串可以看作只读的char数组
        string s1 = "张三";
        string s2 = "张三";

        string s = "abcdefg";
        //s[0] = 'b';  //错误，它是只读的
        Console.WriteLine(s);

        //如果要改变字符串的值
        //先将字符串类型转换为char类型的数组
        char[] chs = s.ToCharArray();
        chs[0] = 'b';  //没问题

        //将字符串数组转为字符串
        s = new string(chs);  //开辟新空间存储新值
        Console.WriteLine(s);

        StringBuilder sb = new StringBuilder();
        string str = null;
        //创建一个计时器，用来记录程序运行时间
        //00:00:00
        Stopwatch sw = new Stopwatch();
        for (int i = 0; i < 100000; i++)
        {
            //str += i;
            sb.Append(i);
        }
        sw.Stop();  //结束计时
        Console.WriteLine(sw.Elapsed);
        //一切类型都可调用ToString
        //Console.WriteLine(sb.ToString());

        Console.ReadKey();


        Console.WriteLine("请输入你喜欢的课程");
        string lessonOne = Console.ReadLine();
        //将字符串转换为大写
        //lessonOne = lessonOne.ToUpper();
        //将字符串转换为小写
        lessonOne = lessonOne.ToLower();
        Console.WriteLine("请输入你喜欢的课程");
        string lessonTwo = Console.ReadLine();
        //将字符串转换为大写
        //lessonTwo = lessonTwo.ToUpper();
        //将字符串转换为小写
        lessonTwo = lessonTwo.ToLower();

        if (lessonOne == lessonTwo)
        {
            Console.WriteLine("你们俩喜欢的课程相同");
        }

        Console.ReadKey();

        string a = "a b   dfd _   +    =  ,,, fdf ";
        //分割字符串 Split
        char[] ch = { ' ', '_', '+', '=', ',' };
        string[] strs = s.Split(ch, StringSplitOptions.RemoveEmptyEntries);
        Console.WriteLine(strs[0]);

        Console.ReadKey();
    }
}
