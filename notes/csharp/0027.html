    F:
        #endregion
        //计算两个整数之间的最大值
        int max = Program.GetMax(1, 3);
        Console.WriteLine(max);
        Console.ReadKey();
    }

    /// <summary>
    /// 计算两个整数之间的最大值，返回最大值
    /// </summary>
    /// <param name="n1">第一个整数</param>
    /// <param name="n2">第二个整数</param>
    /// <returns>将最大值返回</returns>
    public static int GetMax(int n1, int n2)
    {
        return n1 > n2 ? n1 : n2;
    }
}
