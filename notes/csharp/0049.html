14 里氏转换.cs

            
﻿public class Person
{
    public void PersonSayHello()
    {
        Console.WriteLine("我是人类");
    }
}

public class Student : Person
{
    public void StudentSayHello()
    {
        Console.WriteLine("我是学生");
    }
}

public class Teacher : Person
{
    public void TeacherSayHello()
    {
        Console.WriteLine("我是老师");
    }
}

class Prigram
{
    static void Main(string[] args)
    {
        //object 是一切类型的基类
        //里氏转换
        //派生类可以赋值给基类
        //如果基类中存储的是派生类对象
        //那么可以将这个派生类强制转换为派生类类型
        //Student s = new Student();
        //Person p = s;

        string str = string.Join("|", new string[] { "1", "2", "3", "4" });
        Console.WriteLine(str);
        Console.ReadKey();

        Person p = new Student();

        //p中装的是Student类对象
        Student ss = (Student)p;  //正确
        //Teacher t = (Teacher)p;  //错误
        ss.StudentSayHello();
        Console.ReadKey();
        
        //is的用法
        if(p is Teacher)
        {
            Teacher tt = (Teacher)p;
            tt.TeacherSayHello();
        }
        else
        {
            Console.WriteLine("转换失败");
        }
        Console.ReadKey();

        //as的用法
        //如果转换失败tea会变成null
        Teacher tea = p as Teacher;
        //如果转换失败stu会变成null
        Student stu = p as Student;
        Console.ReadKey();
    }
}
