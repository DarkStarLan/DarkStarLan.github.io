﻿//抽象类 不能实例化
public abstract class Animal
{
    //抽象方法 不能有方法体
    //相当于纯虚函数 public void Bark() = 0;
    public abstract void Bark();

    //public void Test()
    //{
    //    //空实现
    //}
}

public class Dog : Animal
{
    public override void Bark()
    {
        Console.WriteLine("狗狗狗狗叫叫叫叫叫叫叫叫");
    }
}

public class Cat : Animal
{
    public override void Bark()
    {
        Console.WriteLine("猫猫猫猫叫叫叫叫叫叫叫叫");
    }
}

class Program
{
    static void Main(string[] args)
    {
        //狗会叫 猫也会叫
        Animal a1 = new Dog();
        a1.Bark();

        Animal a2 = new Cat();
        a2.Bark();

        Console.ReadKey();
    }
}
