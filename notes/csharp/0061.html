﻿using System.Text;

class Program
{
    static void Main(string[] args)
    {
        //FileStream  操作字节
        //使用FileStream读取数据
        FileStream fsRead = new FileStream(@"E:\桌面\test.txt", FileMode.OpenOrCreate, FileAccess.Read);
        byte[] buffer = new byte[1024 * 1024 * 5];
        //返回实际读取到的有效字节数
        int r = fsRead.Read(buffer, 0, buffer.Length);
        //将字节数组中每一个元素解码成字符串
        string s = Encoding.UTF8.GetString(buffer, 0, r);
        //关闭流
        fsRead.Close();
        //释放流占用的资源
        fsRead.Dispose();
        Console.WriteLine(s);
        Console.ReadKey();

        //使用FileStream写入数据
        //使用using，自动关闭和释放流对象
        using (FileStream fsWrite = new FileStream(@"E:\桌面\test.txt", FileMode.OpenOrCreate, FileAccess.Write))
        {
            string str = "!";
            byte[] buffers = Encoding.UTF8.GetBytes(str);
            fsWrite.Write(buffers, 0, buffers.Length);
        }
        Console.ReadKey();

        //StreamReader和StreamWriter  操作字符
    }
}
