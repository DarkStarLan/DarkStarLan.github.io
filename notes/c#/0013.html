﻿class Program
{
    static void Main(string[] args)
    {
        int num = 0;
        bool b = true;
        Console.WriteLine("请输入一个数字");
        try
        {
            num = Convert.ToInt32(Console.ReadLine());
        }
        catch
        {
            Console.WriteLine("输入的内容不能转换为数字");
            b = false;
        }
        if(b)
        {
            Console.WriteLine(num * 2);
        }
        Console.ReadKey();
    }
}
