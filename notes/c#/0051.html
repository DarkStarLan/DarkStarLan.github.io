﻿using System.Collections;

public class Person
{
    public void SayHello()
    {
        Console.WriteLine("我是人类");
    }
}

class Program
{
    static void Main(string[] args)
    {
        //创建集合对象
        ArrayList list = new ArrayList();
        //数组：长度不可变、类型单一
        //集合的好处：长度可变 类型随便
        list.Add(1);
        list.Add(3.14);
        list.Add(true);
        list.Add("张三");
        list.Add('男');
        list.Add(5000m);

        list.Add(new int[] { 1, 2, 3, 4, 5, 6, 7, 8, 9 });
        Person p = new Person();
        list.Add(p);
        list.Add(list);

        //默认情况下打印这个对象
        //自定义类型 打印的是这个类所在的命名空间
        for(int i = 0; i < list.Count; i++)
        {
            if (list[i] is Person)
            {
                ((Person)list[i]).SayHello();
            }
            else if (list[i] is int[])
            {
                for(int j = 0; j < ((int[])list[i]).Length; j++)
                {
                    Console.WriteLine(((int[])list[i])[j]);
                }
            }
            else
            {
                Console.WriteLine(list[i]);
            }
        }
        Console.ReadKey();

        list.Clear();
        list.Add(true);
        list.Add(1);
        list.Add("张三");
        list.AddRange(new int[] { 1, 2, 3, 4, 5, 6, 7, 8, 9 });
        list.AddRange(list);
        for(int i = 0; i < list.Count; i++)
        {
            Console.WriteLine(list[i]);
        }
        Console.ReadKey();

        //根据值删除元素
        list.Remove(true);
        //根据下标删除元素
        list.RemoveAt(0);
        //根据下标和个数删除元素
        list.RemoveRange(0, 2);

        //list.Sort();  //程序不知道怎么排序
        list.Reverse();

        for (int i = 0; i < list.Count; i++)
        {
            Console.WriteLine(list[i]);
        }
        Console.ReadKey();

        list.Clear();

        list.Insert(0, "插入元素");
        list.InsertRange(1, new string[] { "张三", "李四" });
        bool b = list.Contains("张三");
        Console.WriteLine("是否包含: {0}", b);
        for(int i = 0; i < list.Count; i++)
        {
            Console.WriteLine(list[i]);
        }
        Console.ReadKey();
    }
}
