01 类的基本语法.cs

            
﻿public class Person
{
    public string _name;
    public int _age;
    public char _gender;

    public void CHLSS()
    {
        Console.WriteLine("我叫{0}，我今年{1}岁了，我是{2}生，我可以吃喝拉撒睡", this._name, this._age, this._gender);
    }
}

class Program
{
    static void Main(string[] args)
    {
        //创建Person类的对象
        Person sunQuan = new Person();
        sunQuan._name = "孙全";
        sunQuan._age = 23;
        sunQuan._gender = '男';
        sunQuan.CHLSS();
        Console.ReadKey();
    }
}
