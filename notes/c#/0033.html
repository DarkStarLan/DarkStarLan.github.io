﻿class Program
{
    static void Main(string[] args)
    {
        //方法重载和返回值类型没有关系
        int i = M(1, 2);
        double d = M(1.1, 2.2);
        Console.WriteLine(i);
        Console.WriteLine(d);
        Console.ReadKey();
    }

    public static int M(int n1, int n2)
    {
        return n1 + n2;
    }

    public static double M(double d1, double d2)
    {
        return d1 + d2;
    }
