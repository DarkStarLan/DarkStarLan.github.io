29 访问权限修饰符.cs

            
﻿public class Person { }

internal class Student : Person { }

class Program
{
    static void Main(string[] args)
    {
        //C#中有5个访问权限修饰符
        //public        公开权限
        //protected     保护权限
        //private       私有权限
        //internal      只能在当前程序集中访问(默认权限)
        //protected internal
        // -只能在当前程序集中或派生类中访问

        //只有public和internal能修饰class

        //派生类的访问权限高于基类的访问权限会导致
        //基类的成员被暴露，毕竟没有C++的继承方式
    }
}
