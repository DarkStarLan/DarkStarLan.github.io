﻿using System.IO;
using System.Text;

class Program
{
    static void Main(string[] args)
    {
        //StreamReader和StreamWriter

        //读取文本文件
        using (StreamReader sr = new StreamReader(@"E:\桌面\test.txt", Encoding.Default))
        {
            while(!sr.EndOfStream)
            {
                Console.WriteLine(sr.ReadLine());
            }
        }
        Console.ReadKey();

        //写入文本文件
        using (StreamWriter sw = new StreamWriter(@"E:\桌面\new.txt", true))
        {
            sw.Write("123\r\nHello World");
        }
        Console.ReadKey();
    }
}
