﻿using System.Collections;
using System.Diagnostics;

class Program
{
    static void Main(string[] args)
    {
        //装箱和拆箱
        //装箱：将值类型转换为引用类型
        //拆箱：将引用类型转换为值类型
        int n = 10;
        object o = n;  //装箱
        int nn = (int)o;  //拆箱

        //ArrayList list = new ArrayList();
        List<int> list = new List<int>();
        //100万次装箱操作
        Stopwatch sw = new Stopwatch();
        for (int i = 0; i < 1000000; i++)
        {
            list.Add(i);
        }
        sw.Stop();
        Console.WriteLine(sw.Elapsed);
        Console.ReadKey();

        //如果存在继承关系，可能会发生装箱和拆箱
        //如果不存在继承关系，不可能发生装箱和拆箱
        //没有发生装箱和拆箱
        string str = "123";
        int num = Convert.ToInt32(str);

        int number = 10;
        //值类型 转换为 引用类型
        IComparable i = number;  //装箱

        //代码中尽量少进行装箱和拆箱
    }
}
