﻿using System.Collections;

class Program
{
    static void Main(string[] args)
    {
        //Hashtable 哈希表 multimap 字典
        //创建一个键值对集合对象
        //键不能重复，值可以重复
        Hashtable ht = new Hashtable();
        ht.Add(1, "张三");
        ht.Add(2, true);
        ht.Add(3, '男');
        ht.Add(false, "错误的");
        ht[4] = "新来的";  //添加键值对
        ht[1] = "把张三干掉";  //修改值

        if(!ht.ContainsKey(5))  //不存在 键5
        {
            ht.Add(5, "abc");
        }
        else  //存在 键5
        {
            Console.WriteLine("已经存在这个键");
        }

        Console.WriteLine("1\t -> {0}", ht[1]);
        Console.WriteLine("2\t -> {0}", ht[2]);
        Console.WriteLine("3\t -> {0}", ht[3]);
        Console.WriteLine("false\t -> {0}", ht[false]);

        Console.ReadKey();

        //C#是强类型语言  变量类型必须明确定义
        //js是弱类型语言
        var v = ht;
        Console.WriteLine(v.GetType());
        //var 推断类型  类似 auto
        //var 类型变量必须在声明时赋值

        foreach (var item in ht.Keys)
        {
            Console.WriteLine("{0}\t -> {1}", item, ht[item]);
        }

        Console.ReadKey();

        ht.Clear();  //清空元素

        ht.Add(1, "AAA");
        ht.Remove(1);  //根据键删除键值对
        Console.WriteLine("ht的元素个数: {0}", ht.Count);

        Console.ReadKey();
    }
}
