36 接口简介.cs

            
﻿class Person
{
    public void CHLSS()
    {
        Console.WriteLine("我是人类");
    }
}

public class NBAPlayer
{
    public void KouLan()
    {
        Console.WriteLine("扣篮");
    }
}

public interface IKouLanable
{
    void KouLan();
}

public class Student : Person, IKouLanable
{
    public void KouLan()
    {
        Console.WriteLine("我也可以扣篮");
    }
}

public interface IFlyable
{
    void Fly();
    string Test();

    //可以有自动属性
    string Name
    {
        get;
        set;
    }
}

class Program
{
    static void Main(string[] args)
    {
        //接口就是一个规范、能力
        Student s = new Student();
        s.CHLSS();
        s.KouLan();
        Console.ReadKey();

        //接口一般以I...able命名
        //接口中的成员不允许添加访问修饰符
        //默认为public
        //接口中的方法不能用方法体
        //接口中不能包含字段
        //接口中可以有自动属性
    }
}

public class Text
{
    private string _name;
    public string Name
    {
        get { return this._name; }
        set { this._name = value; }
    }

    //自动属性，会自动生成一个private字段
    //和上方的_name一样
    public int Age
    {
        get;
        set;
    }
}
