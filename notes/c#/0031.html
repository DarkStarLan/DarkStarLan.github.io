31 ref参数.cs

            
﻿class Program
{
    static void Main(string[] args)
    {
        //ref 相当于C++中的 引用&
        double salary = 5000;
        JiangJin(ref salary);
        FaKuan(ref salary);
        JiangJin(ref salary);
        Console.WriteLine(salary);
        Console.ReadKey();

        int n1 = 10;
        int n2 = 20;
        //ref要求参数必须在方法外赋值，方法内可以不赋值
        swap(ref n1, ref n2);
        Console.WriteLine(n1);
        Console.WriteLine(n2);
        Console.ReadKey();
    }

    public static void JiangJin(ref double s)
    {
        s += 500;
    }

    public static void FaKuan(ref double s)
    {
        s -= 500;
    }

    public static void swap(ref int n1, ref int n2)
    {
        int temp = n1;
        n1 = n2;
        n2 = temp;
    }
}
