03 静态和非静态.cs

            
﻿public class Person
{
    private static string _name;
    public static string Name
    {
        get { return Person._name; }
        set { Person._name = value; }
    }

    private char _gender;

    public char Gender
    {
        get { return this._gender; }
        set { this._gender = value; }
    }

    public void M1()
    {
        //可以使用静态成员，也可以使用实例(no-static)成员
        Console.WriteLine("非静态的方法");
    }

    public static void M2()
    {
        //只能使用静态(static)成员
        Console.WriteLine("静态的方法");
    }
}

public static class Student
{
    //静态类不能被实例化
    //不能在静态类中声明实例成员
    //private int _age;  //错误
    private static string _name;
    public static string Name
    {
        get { return Student._name; }
        set { Student._name = value; }
    }

    public static void M1()
    {
        Console.WriteLine("Hello World");
    }
}

class Program
{
    static void Main(string[] args)
    {
        //静态函数中只能访问静态成员
        //静态类可以实现"工具类"
        //静态类在整个项目中资源共享
        //静态类占用内存
        //程序全部结束之后，静态类才会被释放
        Person p = new Person();
        p.M1();  //实例方法
        Person.M2();  //静态方法
        p.Gender = '男';
        Console.WriteLine(p.Gender);
        Console.ReadKey();

        //不能创建静态类的对象
        //Student s = new Student();  //错误
        Student.M1();
        Console.ReadKey();
    }
}
