09 Convert类型转换.cs

            
﻿class Program
{
    static void Main(string[] args)
    {
        //当相互转换的类型不兼容时，可使用Convert转换
        string s = "123";
        double d = Convert.ToDouble(s);
        int i = Convert.ToInt32(s);
        Console.WriteLine(d);
        Console.WriteLine(i);
        Console.ReadKey();

        //可用于转换用户输入的字符串
        Console.WriteLine("请输入一个数字");
        string number = Console.ReadLine();
        int num = Convert.ToInt32(number);
        Console.WriteLine("这是{0}，它乘二是{1:0.00}(保留2位小数)", num, num * 2);
        Console.ReadKey();
    }
}
