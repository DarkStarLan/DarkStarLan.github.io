08 类型转换.cs

            
﻿class Program
{
    static void Main(string[] args)
    {
        //隐式类型转换
        int number = 10;
        double d = number;
        Console.WriteLine(d);  //10
        Console.ReadKey();

        //显式类型转换
        double dou = 303.6;
        int n = (int)dou;
        Console.WriteLine(n);  //303
        Console.ReadKey();

        int n1 = 10;
        int n2 = 3;
        double d1 = n1 * 1.0 / n2;  //隐式类型转换
        Console.WriteLine("{0}", d1);
        Console.WriteLine("{0:0.00}", d1);  //保留2位小数
        Console.ReadKey();
    }
}
